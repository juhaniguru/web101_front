Routes

  • POST - api/login

    payload: { "email": "esimerkki@esimerkki.com", "password": "salasana" }

    respose: logged in user

  • POST - api/register

    payload: { "name": "nimi", "email": "esimerkki@esimerkki.com", "password": "salasana" }

    respose: newly registered and logged in user

  • POST - api/user

    payload: NONE

    respose: logged in user

  • GET - api/logout

    payload: NONE

    respose: status 200

  • GET - api/projects

    payload: NONE

    respose: array of projects

  • POST - api/projects

    payload: {"name": "nimi", "description": "kuvaus"}

    respose: newly added project

  • PUT|PATCH - api/projects/{project}

    payload: {"name": "nimi", "description": "kuvaus"}

    respose: newly updated project

  • DELETE - api/projects/{project}

    payload: NONE

    respose: { "destroyed" => true|false }

  • GET - api/projects/{project}/manager

    payload: NONE

    respose: manager of selected project

  • GET - api/projects/{project}/tasks

    payload: NONE

    respose: list of tasks in selected project

  • POST - api/projects/{project}/tasks

    payload: {"name" : "nimi", "description": "kuvaus"}

    respose: newly added task

  • PATCH - api/projects/{project}/tasks/{task}

    payload: {"name" : "nimi", "description": "kuvaus"}

    respose: newly updated task
